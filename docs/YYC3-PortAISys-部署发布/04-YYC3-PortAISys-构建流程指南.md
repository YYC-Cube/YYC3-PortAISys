```typescript
// scripts/build-stats.ts
import fs from 'fs';
import path from 'path';

interface BuildStats {
  totalSize: number;
  totalFiles: number;
  largestFiles: Array<{ file: string; size: number }>;
}

function getBuildStats(outDir: string): BuildStats {
  let totalSize = 0;
  let totalFiles = 0;
  const files: Array<{ file: string; size: number }> = [];

  function walkDir(dir: string) {
    const entries = fs.readdirSync(dir, { withFileTypes: true });

    for (const entry of entries) {
      const fullPath = path.join(dir, entry.name);

      if (entry.isDirectory()) {
        walkDir(fullPath);
      } else {
        const stats = fs.statSync(fullPath);
        const size = stats.size;
        totalSize += size;
        totalFiles++;
        files.push({ file: fullPath, size });
      }
    }
  }

  walkDir(outDir);

  files.sort((a, b) => b.size - a.size);

  return {
    totalSize,
    totalFiles,
    largestFiles: files.slice(0, 10),
  };
}

const stats = getBuildStats('./dist');
console.log('рҹ“Ҡ жһ„е»әз»ҹи®Ў:');
console.log(`жҖ»еӨ§е°Ҹ: ${(stats.totalSize / 1024 / 1024).toFixed(2)} MB`);
console.log(`жҖ»ж–Үд»¶ж•°: ${stats.totalFiles}`);
console.log('жңҖеӨ§ж–Үд»¶:');
stats.largestFiles.forEach(({ file, size }) => {
  console.log(`  ${file}: ${(size / 1024).toFixed(2)} KB`);
});
```

```bash
#!/bin/bash
# scripts/verify-build.sh

# 1. жЈҖжҹҘиҫ“еҮәзӣ®еҪ•
if [ ! -d "dist" ]; then
    echo "вқҢ жһ„е»әиҫ“еҮәзӣ®еҪ•дёҚеӯҳеңЁ"
    exit 1
fi

# 2. жЈҖжҹҘе…ій”®ж–Үд»¶
REQUIRED_FILES=(
    "dist/index.html"
    "dist/assets/index.js"
    "dist/assets/index.css"
)

for file in "${REQUIRED_FILES[@]}"; do
    if [ ! -f "$file" ]; then
        echo "вқҢ зјәе°‘е…ій”®ж–Үд»¶: $file"
        exit 1
    fi
done

# 3. жЈҖжҹҘж–Үд»¶еӨ§е°Ҹ
MAX_SIZE=$((100 * 1024 * 1024))  # 100MB
BUILD_SIZE=$(du -sb dist | cut -f1)

if [ $BUILD_SIZE -gt $MAX_SIZE ]; then
    echo "вҡ пёҸ  иӯҰе‘Ҡ: жһ„е»әеӨ§е°Ҹи¶…иҝҮйҷҗеҲ¶: $(($BUILD_SIZE / 1024 / 1024))MB"
```
